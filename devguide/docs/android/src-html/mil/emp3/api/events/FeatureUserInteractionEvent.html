001package mil.emp3.api.events;
002
003import org.cmapi.primitives.IGeoPosition;
004
005import java.util.EnumSet;
006
007import mil.emp3.api.enums.UserInteractionKeyEnum;
008import mil.emp3.api.enums.UserInteractionMouseButtonEnum;
009import mil.emp3.api.interfaces.IFeature;
010import mil.emp3.api.interfaces.IMap;
011import mil.emp3.api.enums.UserInteractionEventEnum;
012
013/**
014 * This event is generated when the user interacts with features on a specific map.
015 * To register for this event you must instantiate an object that implements the IFeatureInteractionEventListener
016 * interface and register for the event on an IMap, IOverlay or IFeature.
017 */
018public abstract class FeatureUserInteractionEvent extends Event<UserInteractionEventEnum, java.util.List<IFeature>> {
019    private final android.graphics.Point oPoint;
020    private final IGeoPosition oCoordinate;
021    private final IGeoPosition oStartCoordinate;
022    private final UserInteractionMouseButtonEnum oButton;
023    private final EnumSet<UserInteractionKeyEnum> oKeys;
024
025    /**
026     * This constructor is used by the core to generate feature user interaction events other than DRAG events.
027     * @param eEvent The event being evented (Any event except drag).
028     * @param oFeatureList The list of features affected by the event.
029     * @param oPoint The X/Y coordinates of the event.
030     * @param oPosition The geographic coordinate of the event.
031     */
032    protected FeatureUserInteractionEvent(UserInteractionEventEnum eEvent,
033            EnumSet<UserInteractionKeyEnum> keys,
034            UserInteractionMouseButtonEnum button,
035            java.util.List<IFeature> oFeatureList,
036            android.graphics.Point oPoint,
037            IGeoPosition oPosition) {
038        super(eEvent, oFeatureList);
039        this.oPoint = oPoint;
040        this.oCoordinate = oPosition;
041        this.oStartCoordinate = null;
042        this.oButton = button;
043        this.oKeys = keys;
044    }
045
046    /**
047     * This constructor is used by the core to generate feature user interaction drag events.
048     * @param eEvent UserInteractionEventEnum.DRAG
049     * @param oFeatureList The list of features affected by the event.
050     * @param oPoint The X/Y coordinates of the event.
051     * @param oPosition The geographic coordinate of the event end of the drag segment.
052     * @param oStartPosition The geographic coordinate of the start of the drag segment.
053     */
054    protected FeatureUserInteractionEvent(UserInteractionEventEnum eEvent,
055            EnumSet<UserInteractionKeyEnum> keys,
056            UserInteractionMouseButtonEnum button,
057            java.util.List<IFeature> oFeatureList,
058            android.graphics.Point oPoint,
059            IGeoPosition oPosition,
060            IGeoPosition oStartPosition) {
061        super(eEvent, oFeatureList);
062        this.oPoint = oPoint;
063        this.oCoordinate = oPosition;
064        this.oStartCoordinate = oStartPosition;
065        this.oButton = button;
066        this.oKeys = keys;
067    }
068
069    /**
070     * This method returns the x,y coordinate of the location the event occurred at. Where 0,0
071     * is the origin of the map container.
072     * The value of the object is only valid for the duration of the event callback. The event handler
073     * <b>MUST NOT</b> keep a reference to the object.
074     * @return {@android.graphics.Point}
075     */
076    public android.graphics.Point getPoint() {
077        return this.oPoint;
078    }
079
080    /**
081     * This method return the geographic coordinates of the location where the event occurred at.
082     * The value of the coordinate is only valid for the duration of the event callback. The event handler
083     * <b>MUST NOT</b> keep a reference to the coordinate object.
084     * @return An object that implements the IGeoPosition interface, or null if the event occurred off the map.
085     */
086    public IGeoPosition getCoordinate() {
087        return this.oCoordinate;
088    }
089
090    /**
091     * This method returns the object that implements the {@link IMap} interface the event was generated on.
092     * @return {@link IMap}
093     */
094    public abstract IMap getMapEventOccurredOn();
095
096    /**
097     * This method returns the start coordinate. It is only valid for DRAG event. It matches the coordinate
098     * returned by the getCoordindate() of the previous drag event or the coordinate where the down event occured at.
099     * The value of the coordinate is only valid for the duration of the event callback. The event handler
100     * <b>MUST NOT</b> keep a reference to the coordinate object.
101     * @return An object that implements the IGeoPosition interface, or null if the event occurred off the map.
102     */
103    public IGeoPosition getStartCoordinate() {
104        return this.oStartCoordinate;
105    }
106    /**
107     * This method returns the mouse button if pressed.
108     * Values are left, right, middle or none.
109     * @return
110     */
111    public UserInteractionMouseButtonEnum getButton() {
112        return this.oButton;
113    }
114
115    /**
116     * This method returns any keys if pressed
117     * Values are Shift, Ctrl and Alt
118     * @return
119     */
120
121    public EnumSet<UserInteractionKeyEnum> getKeys() {
122        return this.oKeys;
123    }
124}
