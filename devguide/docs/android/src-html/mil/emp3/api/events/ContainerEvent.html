001package mil.emp3.api.events;
002
003import mil.emp3.api.interfaces.IContainer;
004import mil.emp3.api.interfaces.IFeature;
005import mil.emp3.api.interfaces.IMap;
006import mil.emp3.api.interfaces.IOverlay;
007import mil.emp3.api.enums.ContainerEventEnum;
008import org.cmapi.primitives.IGeoBase;
009
010/**
011 * This class implements an abstract Container event. It provides access to all event properties.
012 */
013public abstract class ContainerEvent extends Event<ContainerEventEnum, IContainer> {
014    /**
015     * This constructor is used by the EMP core to create the event.
016     * @param eEvent {@link ContainerEventEnum} This parameter indicates the type of container event.
017     * @param oTarget this parameter indicates the actual container in which the events has occurred.
018     */
019    protected ContainerEvent(ContainerEventEnum eEvent, IContainer oTarget) {
020        super(eEvent, oTarget);
021    }
022
023    /**
024     * This method retrieves a list of the children that the events refers to. If
025     * The event is an ADD, the list contains the children that were added. If its a REMOVE
026     * event the list of children that were removed.
027     * @return 
028     */
029    public abstract java.util.List<IGeoBase> getAffectedChildren();
030
031    /**
032     * This method is a convenience function that indicate if the target is an IMap
033     * @return True if the target object implements the IMap interface. False otherwise.
034     */
035    public boolean targetIsMap() {
036        return (this.getTarget() instanceof IMap);
037    }
038
039    /**
040     * This method is a convenience function that indicate if the target is an IOverlay
041     * @return True if the target object implements the IOverlay interface. False otherwise.
042     */
043    public boolean targetIsOverlay() {
044        return (this.getTarget() instanceof IOverlay);
045    }
046
047    /**
048     * This method is a convenience function that indicate if the target is an IFeature
049     * @return True if the target object implements the IFeature interface. False otherwise.
050     */
051    public boolean targetIsFeature() {
052        return (this.getTarget() instanceof IFeature);
053    }
054
055    /**
056     * This method retrieves the target as an IMap object.
057     * @return An IMap object if the target implements the IMap interface. null otherwise.
058     */
059    public IMap getTargetMap() {
060        if (this.targetIsMap()) {
061            return (IMap) this.getTarget();
062        }
063        return null;
064    }
065
066    /**
067     * This method retrieves the target as an IOverlay object.
068     * @return An IOverlay object if the target implements the IOverlay interface. null otherwise.
069     */
070    public IOverlay getTargetOverlay() {
071        if (this.targetIsOverlay()) {
072            return (IOverlay) this.getTarget();
073        }
074        return null;
075    }
076
077    /**
078     * This method retrieves the target as an IFeature object.
079     * @return An IFeature object if the target implements the IFeature interface. null otherwise.
080     */
081    public IFeature getTargetFeature() {
082        if (this.targetIsFeature()) {
083            return (IFeature) this.getTarget();
084        }
085        return null;
086    }
087}
