001package mil.emp3.api.events;
002
003import mil.emp3.api.enums.LookAtEventEnum;
004import mil.emp3.api.interfaces.ILookAt;
005
006/**
007 * This abstract class implements a lookAT event. It is generated when a lookAt is associated with a map and the map's view changes.
008 */
009public abstract class LookAtEvent extends Event<LookAtEventEnum, ILookAt>{
010    // When apply interface is invoked on the lookAt an event is generated that is handled by map engine to move the camera.
011    // map engine needs to know if lookAt move should be animated. This parameter is of no use to applications.
012
013    private boolean animate;
014
015    /**
016     * This protected constructor is for internal use only.
017     * @param eEvent
018     * @param oLookAt
019     */
020    protected LookAtEvent(LookAtEventEnum eEvent, ILookAt oLookAt, boolean animate) {
021        super(eEvent, oLookAt);
022        this.animate = animate;
023    }
024
025    /**
026     * This method retrieves the target LookAt on which the event was generated.
027     * @return An object that implements the {@link ILookAt} interface.
028     */
029    public abstract ILookAt getLookAt();
030
031    public boolean isAnimate() {
032        return animate;
033    }
034}
