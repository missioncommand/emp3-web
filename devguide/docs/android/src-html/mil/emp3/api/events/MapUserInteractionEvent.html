001package mil.emp3.api.events;
002
003import org.cmapi.primitives.IGeoPosition;
004
005import java.util.EnumSet;
006
007import mil.emp3.api.enums.UserInteractionKeyEnum;
008import mil.emp3.api.enums.UserInteractionMouseButtonEnum;
009import mil.emp3.api.interfaces.IMap;
010import mil.emp3.api.enums.UserInteractionEventEnum;
011
012/**
013 * This event is generated when the user interacts with a specific map.
014 * To register for this event you must instantiate an object that implements the IMapInteractionEventListener
015 * interface and register for the event on an IMap.
016 */
017public abstract class MapUserInteractionEvent extends UserInteractionEvent<IMap> {
018    /**
019     * This constructor is used by the core to generate map user interaction events other than drag.
020     * @param eEvent The event being generated.
021     * @param oMap The map the event occurred on.
022     * @param oPoint The x/y coordinate of the event.
023     * @param oPosition The geographic coordinate of the event.
024     * @param oButton Mouse button, left, right or middle
025     * @param oKeys Keyboard keys, shift, ctrl and/or alt
026     */
027    protected MapUserInteractionEvent(UserInteractionEventEnum eEvent,
028                                      EnumSet<UserInteractionKeyEnum> oKeys,
029                                      UserInteractionMouseButtonEnum oButton,
030                                      IMap oMap,
031                                      android.graphics.Point oPoint,
032                                      IGeoPosition oPosition) {
033        super(eEvent, oKeys, oButton, oMap, oPoint, oPosition);
034    }
035
036    /**
037     * This constructor is used by the core to generate map user interaction drag events.
038     * @param eEvent The event being generated.
039     * @param oMap The map the event occurred on.
040     * @param oPoint The x/y coordinate of the event.
041     * @param oPosition The geographic coordinate of the end of the current drag segment.
042     * @param oStartPosition The geographic coordinate of the start of the drag segment.
043     * @param oButton Mouse button, left, right or middle
044     * @param oKeys Keyboard keys, shift, ctrl and/or alt
045     */
046    protected MapUserInteractionEvent(UserInteractionEventEnum eEvent,
047                                      EnumSet<UserInteractionKeyEnum> oKeys,
048                                      UserInteractionMouseButtonEnum oButton,
049                                      IMap oMap,
050                                      android.graphics.Point oPoint,
051                                      IGeoPosition oPosition,
052                                      IGeoPosition oStartPosition) {
053        super(eEvent, oKeys, oButton, oMap, oPoint, oPosition, oStartPosition);
054    }
055}
