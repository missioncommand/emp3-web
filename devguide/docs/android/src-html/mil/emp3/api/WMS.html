001package mil.emp3.api;
002
003import mil.emp3.api.enums.WMSVersionEnum;
004import mil.emp3.api.exceptions.EMP_Exception;
005import mil.emp3.api.interfaces.core.IStorageManager;
006import mil.emp3.api.interfaces.IWMS;
007import mil.emp3.api.utils.ManagerFactory;
008
009/**
010 * This class defines a WMS service.
011 */
012public class WMS implements IWMS {
013
014    final private IStorageManager storageManager = ManagerFactory.getInstance().getStorageManager();
015
016    private final java.util.UUID uniqueId;
017    private String name = "";
018    private String description = "";
019    private WMSVersionEnum eVersion = WMSVersionEnum.VERSION_1_3;
020    private boolean bTranparent = true;
021    private String sTileFormat = "image/png";
022    private java.util.List<String> oLayers;
023    private final java.net.URL url;
024
025    /**
026     * This constructor creates a WMS
027     * @param sURL This parameter set the url for the service.
028     * @param eWMSVersion This parameter set the WMS version for the service.
029     * @param sTileFormat This parameter sets the format of the tiles requested.
030     * @param bTransparent this parameter set the transparency property.
031     * @param oLayers This is a list of layers that must be displayed.
032     * @throws java.net.MalformedURLException If the url string is invalid.
033     */
034    public WMS(String sURL, 
035            WMSVersionEnum eWMSVersion, 
036            String sTileFormat, 
037            boolean bTransparent, 
038            java.util.List<String> oLayers) 
039            throws java.net.MalformedURLException {
040        
041        uniqueId = java.util.UUID.randomUUID();
042        this.bTranparent = bTransparent;
043        if (oLayers != null) {
044            this.oLayers = oLayers;
045        } else {
046            this.oLayers = new java.util.ArrayList<>();
047        }
048        if ((sTileFormat != null) && !sTileFormat.isEmpty()) {
049            this.sTileFormat = sTileFormat;
050        }
051        if (eWMSVersion != null) {
052            this.eVersion = eWMSVersion;
053        }
054
055        this.url = new java.net.URL(sURL);
056    }
057
058    /**
059     * Set the name for this WMS service.
060     * @param sName the new name that will be set. A value of null sets the name 
061     * to an empty string.
062     */
063    @Override
064    public void setName(String sName) {
065        if (sName == null) {
066            this.name = "";
067        } else {
068            this.name = sName;
069        }
070    }
071
072    /**
073     * This method retrieves the name property.
074     * @return String
075     */
076    @Override
077    public String getName() {
078        return this.name;
079    }
080
081    @Override
082    public java.util.UUID getGeoId() {
083        return this.uniqueId;
084    }
085
086    /**
087     * Set the name for this WMS service.
088     * @param sDescription the new name that will be set. A value of null sets the name 
089     * to an empty string.
090     */
091    @Override
092    public void setDescription(String sDescription) {
093        if (sDescription == null) {
094            this.description = "";
095        } else {
096            this.description = sDescription;
097        }
098    }
099
100    /**
101     * This method retrieves the description property.
102     * @return String
103     */
104    @Override
105    public String getDescription() {
106        return this.description;
107    }
108
109    /**
110     * This method retrieves the URL.
111     * @return java.net.URL
112     */
113    @Override
114    public java.net.URL getURL() {
115        return this.url;
116    }
117
118    /**
119     * This method retrieves the wms version.
120     * @return WMSVersionEnum
121     */    
122    public WMSVersionEnum getWMSVersion() {
123        return this.eVersion;
124    }
125
126    /**
127     * This method retrieves the title format.
128     * @return String
129     */    
130    public String getTileFormat() {
131        return this.sTileFormat;
132    }
133
134    /**
135     * This method retrieves the transparency property.
136     * @return String
137     */    
138    public boolean getTransaparent() {
139        return this.bTranparent;
140    }
141
142    /**
143     * This method sets the layer list. The updated list is sent to all maps utilizing
144     * this service.
145     * @param newLayers A list of string. The list is ignored if its null or empty.
146     * @throws EMP_Exception 
147     */
148    @Override
149    public void setLayers(java.util.List<String> newLayers) throws EMP_Exception {
150        if ((newLayers != null) && !newLayers.isEmpty()) {
151            this.oLayers = newLayers;
152            storageManager.MapServiceUpdated(this);
153        }
154    }
155
156    /**
157     * This method retrieves the current layer list.
158     * @return A list of String values.
159     */
160    @Override
161    public java.util.List<String> getLayers() {
162        return this.oLayers;
163    }
164}
