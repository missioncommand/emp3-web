001package mil.emp3.api.listeners;
002
003
004import mil.emp3.api.interfaces.IEvent;
005
006/**
007 * This interface defines the base of all event listener interfaces.
008 * @param <T> An object the is derived from {@link IEvent}.
009 */
010public interface IEventListener<T extends IEvent> {
011
012  /** 
013   * This method must be implemented by an object class to handle T type events.
014   * @param event An event class that implements the {@link IEvent} interface.
015   */
016  public void onEvent(T event);
017}
