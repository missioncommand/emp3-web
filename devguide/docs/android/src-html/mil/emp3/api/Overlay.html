001package mil.emp3.api;
002
003
004import org.cmapi.primitives.GeoContainer;
005import org.cmapi.primitives.IGeoBase;
006import org.cmapi.primitives.IGeoContainer;
007
008import java.util.List;
009
010import mil.emp3.api.abstracts.Container;
011import mil.emp3.api.exceptions.EMP_Exception;
012import mil.emp3.api.interfaces.IFeature;
013import mil.emp3.api.interfaces.IOverlay;
014import mil.emp3.api.interfaces.core.IStorageManager;
015import mil.emp3.api.utils.ManagerFactory;
016
017/**
018 * This class handles all operations on a Geo Container object or Overlay.
019 */
020public class Overlay extends Container implements IOverlay {
021
022    final private IStorageManager storageManager = ManagerFactory.getInstance().getStorageManager();
023
024    /**
025     * This default constructor creates an Overlay and its encapsulated GeoContainer.
026     */
027    public Overlay() {
028        super(new GeoContainer());
029    }
030
031    /**
032     * This constructor create the overlay object with the provided object encapsulated within.
033     * @param geoContainer and object that implements the IGeoContainer interface.
034     */
035    public Overlay(IGeoContainer geoContainer) {
036        super(geoContainer);
037    }
038
039    @Override
040    public java.util.List<IOverlay> getOverlays() {
041        return storageManager.getChildOverlays(this);
042    }
043
044    @Override
045    public void addOverlay(IOverlay overlay, boolean visible)
046            throws EMP_Exception {
047        if (overlay == null) {
048            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "Parameter to Overlay.addOverlay can not be null.");
049        }
050        
051        java.util.ArrayList<IOverlay> oList = new java.util.ArrayList<>();
052        oList.add(overlay);
053        this.addOverlays(oList, visible);
054    }
055
056    @Override
057    public void addOverlays(java.util.List<IOverlay> overlays, boolean visible)
058            throws EMP_Exception {
059        if (overlays == null) {
060            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "Parameter to Overlay.addOverlays can not be null.");
061        } else if (overlays.size() > 0) {
062            storageManager.addOverlays(this, overlays, visible);
063        }
064    }
065
066    @Override
067    public void removeOverlay(IOverlay overlay)
068            throws EMP_Exception {
069        if(null == overlay) return;
070        java.util.ArrayList<IOverlay> oList = new java.util.ArrayList<>();
071        oList.add(overlay);
072        this.removeOverlays(oList);
073    }
074
075    @Override
076    public void removeOverlays(java.util.List<IOverlay> overlays)
077            throws EMP_Exception {
078        if((null == overlays) || (0 == overlays.size())) return;
079        storageManager.removeOverlays(this, overlays);
080    }
081
082    @Override
083    public java.util.List<IFeature> getFeatures() {
084        return storageManager.getChildFeatures(this);
085    }
086
087    @Override
088    public void addFeature(IFeature feature, boolean visible)
089            throws EMP_Exception {
090        if (feature == null) {
091            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "Parameter to Overlay.addFeature can not be null.");
092        }
093        
094        java.util.ArrayList<IFeature> oList = new java.util.ArrayList<>();
095        oList.add(feature);
096        this.addFeatures(oList, visible);
097    }
098
099    @Override
100    public void addFeatures(java.util.List<IFeature> features, boolean visible)
101            throws EMP_Exception {
102        if (features == null) {
103            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "Parameter to Overlay.addFeatures can not be null.");
104        } else if (features.size() > 0) {
105            storageManager.addFeatures(this, features, visible);
106        }
107    }
108
109    @Override
110    public void removeFeature(IFeature feature)
111            throws EMP_Exception {
112        if(null == feature) return;
113        java.util.ArrayList<IFeature> oList = new java.util.ArrayList<>();
114        oList.add(feature);
115        this.removeFeatures(oList);
116    }
117
118    @Override
119    public void removeFeatures(java.util.List<IFeature> features)
120            throws EMP_Exception {
121        if((null == features) || (0 == features.size())) return;
122        storageManager.removeFeatures(this, features);
123    }
124
125    @Override
126    public void apply() {
127
128    }
129}
