001package mil.emp3.api;
002
003import mil.emp3.api.abstracts.Feature;
004
005import org.cmapi.primitives.GeoCircle;
006import org.cmapi.primitives.IGeoCircle;
007
008import mil.emp3.api.enums.FeatureTypeEnum;
009
010/**
011 * This class implements a Circle feature. It requires a single coordinate that indicates the
012 * geographic position of the center of the circle.
013 */
014public class Circle extends Feature implements IGeoCircle {
015
016    /**
017     * This constructor creates a Circle feature with a default radius.
018     */
019    public Circle() {
020        super(new GeoCircle(), FeatureTypeEnum.GEO_CIRCLE);
021    }
022
023    /**
024     * This constructor creates a circle with the indicated radius.
025     * @param radius The required radius in meters. If the radius is < 0 the absolute value is used. If the value is == 0, the default radius is used.
026     */
027    public Circle(double radius) {
028        super(new GeoCircle(), FeatureTypeEnum.GEO_CIRCLE);
029        if (radius < 0) {
030            radius = Math.abs(radius);
031        } else if (radius == 0) {
032            return;
033        }
034        this.getRenderable().setRadius(radius);
035    }
036
037    /**
038     * This constructor create a circle feature based on the GeoCircle object provided.
039     * @param renderable see IGeoCircle. If the GeoCircle does not have a radius it is set to the default radius.
040     */
041    public Circle(IGeoCircle renderable) {
042        super(renderable, FeatureTypeEnum.GEO_CIRCLE);
043    }
044
045    /**
046     * This method returns a reference to the GeoCircle object encapsulated in the feature.
047     * @return See {@link IGeoCircle}
048     */
049    @Override
050    public final IGeoCircle getRenderable() {
051        return (IGeoCircle) super.getRenderable();
052    }
053
054    /**
055     * This method overrides the default radius of a circular feature.
056     * @param radius The new radius in meters. If the radius is < 0 the absolute value is used. If the value is == 0, the radius is NOT set.
057     */
058    @Override
059    public void setRadius(double radius) {
060        if (radius < 0) {
061            radius = Math.abs(radius);
062        } else if (radius == 0) {
063            return;
064        }
065        this.getRenderable().setRadius(radius);
066    }
067
068    /**
069     * This method retrieves the current radius setting.
070     * @return The current radius.
071     */
072    @Override
073    public double getRadius() {
074        return this.getRenderable().getRadius();
075    }
076}
