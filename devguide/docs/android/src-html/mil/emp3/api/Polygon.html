001package mil.emp3.api;
002
003
004import org.cmapi.primitives.GeoPolygon;
005import org.cmapi.primitives.IGeoPolygon;
006import org.cmapi.primitives.IGeoPosition;
007
008import java.util.List;
009import mil.emp3.api.abstracts.Feature;
010
011import mil.emp3.api.enums.FeatureTypeEnum;
012
013/**
014 * This class implements an polygon. It requires 3 or more geographic coordinates.
015 */
016public class Polygon extends Feature implements IGeoPolygon {
017
018    public Polygon() {
019        super(new GeoPolygon(), FeatureTypeEnum.GEO_POLYGON);
020    }
021
022    public Polygon(List<IGeoPosition> oPositionList) {
023        super(new GeoPolygon(), FeatureTypeEnum.GEO_POLYGON);
024        this.setPositions(oPositionList);
025    }
026
027    public Polygon(IGeoPolygon oRenderable) {
028        super(oRenderable, FeatureTypeEnum.GEO_POLYGON);
029    }
030
031    /**
032     * This method returns a reference to the GeoPolygon object encapsulated in the feature.
033     * @return See {@link IGeoPolygon}
034     */
035    @Override
036    public final IGeoPolygon getRenderable() {
037        return (IGeoPolygon) super.getRenderable();
038    }
039}
