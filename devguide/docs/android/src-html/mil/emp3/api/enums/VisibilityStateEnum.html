001package mil.emp3.api.enums;
002
003/**
004 *
005 * This class enumerates the visibility states of an object. The visibility 
006 * state of an object is specific to a specific map. The same object can have different
007 * visibility states on different maps.
008 */
009public enum VisibilityStateEnum {
010    /**
011     * This enumerated value indicates that the object is visible on the map. The
012     * visibility of any object contained within the object will depend on its own
013     * visibility state on the same map.
014     */
015    VISIBLE,
016
017    /**
018     * This enumerated value is only relevant for the visibility state of an object
019     * in reference to a specific parent on a specific map. (See {@link } It 
020     * indicates that the object is marked visible under the parent in question but 
021     * is hidden due to the parents ancestors (that are on the same map) visibility 
022     * setting. The object would not be visible if it had no other parents on the 
023     * same map. If the object had other parents on the same map the objects 
024     * visibility on the map would depend on its visibility state with those other parents.
025     */
026    VISIBLE_ANCESTOR_HIDDEN,
027
028    /* 
029    Not visible in relation to parent.  In the case of Map.getVisible(target) then HIDDEN means it is not visible to user on map engine at all.  HIDDEN in relation to a specific parent does not guarantee that it is not visible to the user under a different parent
030    */
031    HIDDEN
032}
