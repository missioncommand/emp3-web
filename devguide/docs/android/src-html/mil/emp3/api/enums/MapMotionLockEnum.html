001package mil.emp3.api.enums;
002
003/**
004 * This class enumerates the different locks that can be placed on the map's motion.
005 */
006public enum MapMotionLockEnum {
007
008    /**
009     * This enumerated value indicates that the map view is free to change i.e. pan, zoom, rotate, tilt
010     * based on Gestures. Note that pan, zoom, rotate, tilt is not necessarily an exhaustive list of motions
011     */
012    UNLOCKED,
013    /**
014     * This enumerated value indicates that the map will not pan, zoom or rotate, tilt based on Gestures. The only
015     * motion allowed is panning if the user is dragging an object at the edge of the viewing area.
016     * This is currently used by the Freehand Editor
017     */
018    SMART_LOCK,
019    /**
020     * This enumerated value indicates that the map will not pan, zoom, rotate, tilt based on Gestures.
021     * Client application can still use Camera and LookAt to change the map view.
022     */
023    LOCKED
024}
