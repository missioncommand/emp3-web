001package mil.emp3.api.enums;
002
003/**
004 * This class enumerates the different MilStd label settings.
005 */
006public enum MilStdLabelSettingEnum {
007    /**
008     * This value indicates that the map engine shall display required MilStd labels only.
009     * V, L, S, AA, AB, AC
010     */
011    REQUIRED_LABELS,
012    /**
013     * This value indicates that the map engine shall display the common MilStd labels.
014     * In addition to the required labels.
015     * H, M, T, T1, CN
016     */
017    COMMON_LABELS,
018    /**
019     * This value indicates that the map engine shall display all MilStd labels.
020     * In addition to the common labels.
021     * C, F, G, H1, H2, J, K, N, P, W, W1, X, Y, Z
022     */
023    ALL_LABELS
024}
