001package mil.emp3.api.enums;
002
003/**
004 * This class enumerates the MilStd affiliation values.
005 */
006public enum MilStd2525Affiliation {
007    PENDING,
008    UNKNOWN,
009    FRIEND,
010    NEUTRAL,
011    HOSTILE,
012    ASSUMED_FRIEND,
013    SUSPECT,
014    EXERCISE_PENDING,
015    EXERCISE_UNKNOWN,
016    EXERCISE_FRIEND,
017    EXERCISE_NEUTRAL,
018    EXERCISE_ASSUMED_FRIEND,
019    JOKER,
020    FAKER;
021
022    @Override
023    public String toString() {
024        switch (this) {
025            case PENDING:
026                return "P";
027            case UNKNOWN:
028                return "U";
029            case FRIEND:
030                return "F";
031            case NEUTRAL:
032                return "N";
033            case HOSTILE:
034                return "H";
035            case ASSUMED_FRIEND:
036                return "A";
037            case SUSPECT:
038                return "S";
039            case EXERCISE_PENDING:
040                return "G";
041            case EXERCISE_UNKNOWN:
042                return "W";
043            case EXERCISE_FRIEND:
044                return "D";
045            case EXERCISE_NEUTRAL:
046                return "L";
047            case EXERCISE_ASSUMED_FRIEND:
048                return "M";
049            case JOKER:
050                return "J";
051            case FAKER:
052                return "K";
053        }
054        
055        return "*";
056    }
057}
