001package mil.emp3.api.enums;
002
003import mil.emp3.api.interfaces.IEventEnum;
004/**
005 * This class enumerates the user interaction events that can be generated by a map.
006 */
007public enum UserInteractionEventEnum implements IEventEnum {
008    /**
009     * An event indicating that an object has been clicked.
010     */
011    CLICKED,
012
013    /**
014     * An event indicating that an object has been double clicked.
015     */
016    DOUBLE_CLICKED,
017
018    /**
019     * An event indicating that an object has been long pressed.
020     */
021    LONG_PRESS,
022    //PRESS_RELEASE,
023
024    /**
025     * This value indicates that a feature or map has been dragged.
026     */
027    DRAG,
028    /**
029     * This enumerated value indicates that the drag operation has completed.
030     */
031    DRAG_COMPLETE
032}
