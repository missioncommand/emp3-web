001package mil.emp3.api.enums;
002
003import mil.emp3.api.interfaces.IEventEnum;
004
005/**
006 * This class enumerates the different events that can be generated by a camera.
007 */
008public enum CameraEventEnum implements IEventEnum {
009    /**
010     * This enumerated value indicates that the target camera is currently in motion. One or more camera settings have been changed.
011     */
012    CAMERA_IN_MOTION,
013
014    /**
015     * This enumerated value indicates that the target camera has stopped moving. The event contains the cameras final settings.
016     */
017    CAMERA_MOTION_STOPPED
018}
