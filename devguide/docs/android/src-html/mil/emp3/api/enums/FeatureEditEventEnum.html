001package mil.emp3.api.enums;
002
003import mil.emp3.api.interfaces.IEventEnum;
004/**
005 * This class enumerates the type of feature edit events
006 */
007public enum FeatureEditEventEnum implements IEventEnum {
008    /**
009     * This value indicates that the edit has started. The event target provides the feature being edited.
010     */
011    EDIT_START,
012    /**
013     * This value indicates that the feature has been changed. The event target is a reference to the updated feature.
014     */
015    EDIT_UPDATE,
016    /**
017     * This value indicates that the edit has been canceled. The event target is a reference of the feature prior to the edit.
018     */
019    EDIT_CANCELED,
020    /**
021     * This value indicates that the edit has completed. The event target is a reference of the edited feature.
022     */
023    EDIT_COMPLETE
024}
