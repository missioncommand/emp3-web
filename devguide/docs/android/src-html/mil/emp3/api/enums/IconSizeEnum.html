001package mil.emp3.api.enums;
002
003/**
004 * This class enumerate the icon sizes suported by the system. Actual pixel size is not used do
005 * to the difference in pixel density of android devices.
006 */
007public enum IconSizeEnum {
008    /**
009     * This enumerated value indicates the smallest size possible. It should match 0.6 the size of a SMALL icon.
010     */
011    TINY,
012    /**
013     * This enumerated value indicates a size equivalent to a 1/3 in X 1/3 in icon. Which on a standard 96dpi screen is ~32x32.
014     */
015    SMALL,
016    /**
017     * This enumerated value indicates a size equivalent to 1.4 the size of a SMALL icon.
018     */
019    MEDIUM,
020    /**
021     * This enumerated value indicates a size equivalent to 1.8 the size of a SMALL icon.
022     */
023    LARGE;
024
025    /**
026     * This method return the scaling factor associated with the enumerated value.
027     * @return scale factor.
028     */
029    public double getScaleFactor() {
030        double dRet = 1.0;
031
032        switch (this) {
033            case TINY:
034                dRet = 0.7; // 70% of the original size.
035                break;
036            case SMALL:
037                dRet = 1.0; // 100% of the original size.
038                break;
039            case MEDIUM:
040                dRet = 1.4; // 140% of the original size.
041                break;
042            case LARGE:
043                dRet = 1.8; // 180% of the original size.
044                break;
045        }
046
047        return dRet;
048    }
049}
