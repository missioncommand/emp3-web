001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoContainer;
004
005import mil.emp3.api.exceptions.EMP_Exception;
006
007/**
008 * This interface defines the the methods associated with an overlay.
009 */
010public interface IOverlay extends IContainer, IGeoContainer {
011    /**
012     * This method retrieves a list of this overlay's child overlays.
013     *
014     * @return An ArrayList of zero or more IOverlay objects.
015     */
016    public java.util.List<IOverlay> getOverlays();
017
018    /**
019     * This method adds a child overlay to this overlay.
020     *
021     * @param overlay The child overlay to add.
022     * @param visible True if the overlay is to be made visible, false otherwise.
023     * @throws EMP_Exception The exception is raised if a processing error is encountered.
024     */
025    public void addOverlay(IOverlay overlay, boolean visible)
026            throws EMP_Exception;
027
028    /**
029     * This method adds one or more child overlays to this overlay.
030     *
031     * @param overlays A list of child overlays to add.
032     * @param visible True if the overlays are to be made visible, false otherwise.
033     * @throws EMP_Exception The exception is raised if a processing error is encountered.
034     */
035    public void addOverlays(java.util.List<IOverlay> overlays, boolean visible)
036            throws EMP_Exception;
037
038    /**
039     * This method removes a child overlay from this overlay.
040     *
041     * @param overlay The child overlay to remove.
042     * @throws EMP_Exception The exception is raised if a processing error is encountered.
043     */
044    public void removeOverlay(IOverlay overlay)
045            throws EMP_Exception;
046
047    /**
048     * This method removes one or more child overlays from this overlay.
049     *
050     * @param overlays A list of child overlays to remove.
051     * @throws EMP_Exception The exception is raised if a processing error is encountered.
052     */
053    public void removeOverlays(java.util.List<IOverlay> overlays)
054            throws EMP_Exception;
055
056    /**
057     * This method retrieves all child features of this overlay.
058     * @return An ArrayList of IFeatures.
059     */
060    public java.util.List<IFeature> getFeatures();
061
062    /**
063     * This method adds a feature to this overlay.
064     * @param feature The feature to add
065     * @param visible True if the feature is to be made visible, false otherwise.
066     * @throws EMP_Exception The exception is raised if a processing error is encountered.
067     */
068    public void addFeature(IFeature feature, boolean visible)
069            throws EMP_Exception;
070
071    /**
072     * This method adds one or more features to this overlay.
073     * @param features ArrayList of IFeatures to add.
074     * @param visible True if the features are to be made visible, false otherwise.
075     * @throws EMP_Exception The exception is raised if a processing error is encountered.
076     */
077    public void addFeatures(java.util.List<IFeature> features, boolean visible)
078            throws EMP_Exception;
079
080    /**
081     * This method removes a child feature from this overlay.
082     *
083     * @param feature The child feature to remove.
084     * @throws EMP_Exception The exception is raised if a processing error is encountered.
085     */
086    public void removeFeature(IFeature feature)
087            throws EMP_Exception;
088
089    /**
090     * This method removes one or more child features from this overlay.
091     *
092     * @param features A list of child features to remove.
093     * @throws EMP_Exception The exception is raised if a processing error is encountered.
094     */
095    public void removeFeatures(java.util.List<IFeature> features)
096            throws EMP_Exception;
097
098    /**
099     * This method triggers an update of the overlay. After changing any properties, this
100     * method MUST be called to force an update.
101     */
102    public void apply();
103}
