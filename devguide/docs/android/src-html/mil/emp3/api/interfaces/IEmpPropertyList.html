001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoColor;
004
005/**
006 * This class defines the interface to an EmpPropertyList object.
007 */
008public interface IEmpPropertyList extends java.util.Map<String, Object> {
009    /**
010     * This method is used internally to generate an exception if the key is not present on the list.
011     * @param key
012     * @return
013     */
014    boolean genExceptionIfNotPresent(String key);
015
016    /**
017     * This method return the string value of the entry identified by the key provided. An exception
018     * is generated if the value contained in the list is not a String.
019     * @param key
020     * @return
021     */
022    String getStringValue(String key);
023
024    /**
025     * This method return the IGeoColor object from the value of the entry identified by the key provided. An exception
026     * is generated if the value contained in the list is not a IGeoColor object.
027     * @param key
028     * @return
029     */
030    IGeoColor getColorValue(String key);
031
032    /**
033     * This method return the double value from the value of the entry identified by the key provided. An exception
034     * is generated if the value contained in the list is not a double.
035     * @param key
036     * @return
037     */
038    double getDoubleValue(String key);
039
040    mil.emp3.api.enums.MirrorCacheModeEnum getMirrorCacheModeEnum(String key);
041
042    android.content.Context getContext(String key);
043}
