001package mil.emp3.api;
002
003
004import org.cmapi.primitives.GeoPoint;
005import org.cmapi.primitives.IGeoIconStyle;
006import org.cmapi.primitives.IGeoPoint;
007import org.cmapi.primitives.IGeoPosition;
008
009import mil.emp3.api.enums.FeatureTypeEnum;
010import mil.emp3.api.abstracts.Feature;
011import org.cmapi.primitives.GeoIconStyle;
012import org.cmapi.primitives.GeoPosition;
013
014/**
015 * This class implements the Point feature that encapsulates a GeoPoint object.
016 */
017public class Point extends Feature implements IGeoPoint {
018    private double dIconScale = 1.0;
019    private int resourceId = 0;
020
021    public Point() {
022        super(new GeoPoint(), FeatureTypeEnum.GEO_POINT);
023    }
024
025    public Point(String sURL) {
026        super(new GeoPoint(), FeatureTypeEnum.GEO_POINT);
027        this.setIconURI(sURL);
028    }
029
030    public Point(double dLat, double dLong) {
031        super(new GeoPoint(), FeatureTypeEnum.GEO_POINT);
032        
033        IGeoPosition oPos = new GeoPosition();
034        oPos.setLatitude(dLat);
035        oPos.setLongitude(dLong);
036        this.setPosition(oPos);
037    }
038
039    public Point(IGeoPoint oRenderable) {
040        super(oRenderable, FeatureTypeEnum.GEO_POINT);
041    }
042
043    /**
044     * This method returns a reference to the GeoPoint object encapsulated in the feature.
045     * @return See {@link IGeoPoint}
046     */
047    @Override
048    public final IGeoPoint getRenderable() {
049        return (IGeoPoint) super.getRenderable();
050    }
051
052    /**
053     * This method set the image style.
054     * @param oStyle See {@link IGeoIconStyle}
055     */
056    @Override
057    public void setIconStyle(IGeoIconStyle oStyle) {
058        ((IGeoPoint) this.getRenderable()).setIconStyle(oStyle);
059    }
060
061    /**
062     * This method retrieves the icon style of the point if one has been set.
063     * @return An {@link IGeoIconStyle} or null
064     */
065    @Override
066    public IGeoIconStyle getIconStyle() {
067        return this.getRenderable().getIconStyle();
068    }
069
070    /**
071     * This method set the image URL of a point feature. It will override the default icon.
072     * @param sURL The URL of the image.
073     */
074    @Override
075    public void setIconURI(String sURL) {
076        ((IGeoPoint) this.getRenderable()).setIconURI(sURL);
077    }
078
079    /**
080     * This method retrieves the URL of the image
081     * @return URL string.
082     */
083    @Override
084    public String getIconURI() {
085        return ((IGeoPoint) this.getRenderable()).getIconURI();
086    }
087
088    /**
089     * This method sets the icon scale. The size of the icon will be modified by this factor when rendered.
090     * @param dScale A value smaller than 1.0 decreases the size. A value larger than 1.0 increases the size.
091     */
092    public void setIconScale(double dScale) {
093        this.dIconScale = Math.abs(dScale);
094    }
095
096    /**
097     * This method returns the scale value.
098     * @return
099     */
100    public double getIconScale() {
101        return this.dIconScale;
102    }
103
104    /**
105     * This method is an ANDROID ONLY method. It allows the developer to us an android drawable resource
106     * as an icon for a Point feature. The resource must be accessible to the EMP Core. If a
107     * resource Id is set the icon style offset is interpreted as a fraction.
108     * @param resId The resource ID of the resource.
109     */
110    public void setResourceId(int resId) {
111        this.resourceId = resId;
112    }
113
114    /**
115     * This method is an ANDROID ONLY method. It returns the resource Id.
116     * @return The resojurce Id or 0 if one has not been set.
117     */
118    public int getResourceId() {
119        return this.resourceId;
120    }
121}
