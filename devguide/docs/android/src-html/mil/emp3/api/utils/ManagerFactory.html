001package mil.emp3.api.utils;
002
003import android.app.Application;
004import android.content.Context;
005import android.content.pm.ApplicationInfo;
006import android.util.Log;
007
008import java.io.File;
009
010import dalvik.system.DexClassLoader;
011import mil.emp3.api.interfaces.core.ICoreManager;
012import mil.emp3.api.interfaces.core.IEventManager;
013import mil.emp3.api.interfaces.core.IStorageManager;
014import mil.emp3.mapengine.CopySharedObjectFile;
015import mil.emp3.mapengine.interfaces.IMapInstance;
016import mil.emp3.mapengine.interfaces.IMilStdRenderer;
017
018public class ManagerFactory {
019    static private String TAG = ManagerFactory.class.getSimpleName();
020
021    static private ManagerFactory instance;
022
023    final private IStorageManager storageManager;
024    final private IEventManager eventManager;
025    final private ICoreManager coreManager;
026
027    final private IMilStdRenderer milStdRenderer;
028
029    static final private Application application;
030
031    // All classes are loaded from the same APK so use one dexLoader for all managers. Actually it is a must otherwise
032    // Classes created by these managers will belong to different class loaders and to begin with single instance logic
033    // for those classes will fail. Example is BitmapCachefactory.
034
035    private DexClassLoader dexLoader = null;
036
037    static {
038        try { // retrieves the Application object
039            application = (Application) Class.forName("android.app.ActivityThread").getMethod("currentApplication").invoke(null, (Object[]) null);
040        } catch (Exception e) {
041            throw new IllegalStateException("ERROR: " + e.getMessage(), e);
042        }
043    }
044
045    private ManagerFactory() {
046        try {
047            storageManager = (IStorageManager) dexLoadClass("mil.emp3.emp3_android_sdk_core_apk", "mil.emp3.core.storage.StorageManager");
048            eventManager   = (IEventManager)   dexLoadClass("mil.emp3.emp3_android_sdk_core_apk", "mil.emp3.core.EventManager");
049            coreManager    = (ICoreManager)    dexLoadClass("mil.emp3.emp3_android_sdk_core_apk", "mil.emp3.core.CoreManager");
050
051            milStdRenderer = (IMilStdRenderer) dexLoadClass("mil.emp3.emp3_android_sdk_core_apk", "mil.emp3.core.utils.MilStdRenderer");
052
053
054            storageManager.setEventManager(eventManager);
055            eventManager.setStorageManager(storageManager);
056            coreManager.setStorageManager(storageManager);
057            coreManager.setEventManager(eventManager);
058
059            milStdRenderer.setStorageManager(storageManager);
060
061        } catch (ClassNotFoundException | InstantiationException | IllegalAccessException e) {
062            throw new IllegalStateException("Unable to initialize ManagerFactory: " + e.getMessage(), e);
063        }
064    }
065
066    static public ManagerFactory getInstance() {
067        if (instance == null) {
068            synchronized (ManagerFactory.class) {
069                if (instance == null) {
070                    instance = new ManagerFactory();
071                }
072            }
073        }
074        return instance;
075    }
076
077    static private Context getContext() {
078        return application.getApplicationContext();
079    }
080
081    public IStorageManager getStorageManager() {
082        return storageManager;
083    }
084
085    public IEventManager getEventManager() {
086        return eventManager;
087    }
088
089    public ICoreManager getCoreManager() {
090        return coreManager;
091    }
092
093    public IMilStdRenderer getMilStdRenderer() {
094        return milStdRenderer;
095    }
096
097    /**
098     * If application compiled in the aar then class will be loaded locally, else it wil be loaded from the apk.
099     *     compile (group: 'mil.army.sec.smartClient', name: 'emp3-android-sdk-core', version: "$version_emp3Android", ext: 'aar') { transitive = true }
100     *     OR
101     *     adb install emp3-android-sdk-core-version.apk
102     * @param apkName
103     * @param className
104     * @return
105     * @throws ClassNotFoundException
106     * @throws InstantiationException
107     * @throws IllegalAccessException
108     */
109    private Object dexLoadClass(String apkName, String className) throws ClassNotFoundException, InstantiationException, IllegalAccessException {
110        Log.d(TAG, "dexLoadClass: " + className);
111
112        try {
113            // Attempt to load locally, this will be the case for st5and-alone application
114            Class<?> clazz = (Class<IMapInstance>) Class.forName(className, true, getClass().getClassLoader());
115            return clazz.newInstance();
116        } catch (ClassNotFoundException e) {
117            Log.i(TAG, className + " is not part of the host application, will attempt to load from the standalone APK ");
118        }
119
120        if(null == dexLoader) {
121            final File dexOutputDir = getContext().getDir("dex" + className, Context.MODE_PRIVATE);
122            Log.d(TAG, "dexOutputDir: " + dexOutputDir);
123
124            final String apkPath = getApkPath(apkName);
125            Log.d(TAG, "apkPath: " + apkPath);
126
127            if (apkPath != null) {
128                dexLoader = new DexClassLoader(apkPath, dexOutputDir.getAbsolutePath(), CopySharedObjectFile.getNativeLibDir(getContext(), apkName), getContext().getClassLoader());
129                return dexLoader.loadClass(className).newInstance();
130
131            } else {
132                throw new IllegalStateException("ERROR: " + apkName + " not found. Did you install it?");
133            }
134        } else {
135            return dexLoader.loadClass(className).newInstance();
136        }
137    }
138
139    private String getApkPath(String apkName) {
140        Log.d(TAG, "getApkPath: " + apkName);
141
142        for (ApplicationInfo app : getContext().getPackageManager().getInstalledApplications(0)) {
143            if (!((app.flags & ApplicationInfo.FLAG_UPDATED_SYSTEM_APP) == ApplicationInfo.FLAG_UPDATED_SYSTEM_APP)) {
144                if (!((app.flags & ApplicationInfo.FLAG_SYSTEM) == ApplicationInfo.FLAG_SYSTEM)) {
145                    final String uri = app.sourceDir;
146                    if (uri.contains(apkName)) {
147                        return uri;
148                    }
149                }
150            }
151        }
152        return null;
153    }
