001package mil.emp3.api.exceptions;
002
003/**
004 * This exception class is the base of the EMP V3 exceptions.
005 */
006public class EMP_Exception extends Exception {
007    public enum ErrorDetail {
008        INVALID_PARAMETER,
009        INVALID_MAP,
010        INVALID_PARENT,
011        INVALID_CHILD,
012        NOT_SUPPORTED,
013        OTHER
014    }
015    
016    private final ErrorDetail eErrorDeatil;
017    
018    public EMP_Exception(EMP_Exception.ErrorDetail detailEnum, String errorMessage) {
019        super(errorMessage);
020        this.eErrorDeatil = detailEnum;
021    }
022    
023    public EMP_Exception.ErrorDetail getErrorDeatil() {
024        return this.eErrorDeatil;
025    }
026}
