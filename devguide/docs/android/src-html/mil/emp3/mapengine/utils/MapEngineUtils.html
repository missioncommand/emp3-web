039    /**
040     * This method returns a stripple pattern for the given stroke stype pattern.
041     * @param strokeStyle
042     * @return
043     */
044    public static short getStipplePattern(IGeoStrokeStyle strokeStyle) {
045        short pattern = 0;
046
047        switch (strokeStyle.getStrokePattern()) {
048            case solid:
049                pattern = 0;
050                break;
051            case dashed:
052                pattern = (short) 0xE73C;
053                break;
054            case dotted:
055                pattern = (short) 0xAAAA;
056                break;
057        }
058
059        return pattern;
060    }

113    public static int getTGStippleFactor(String basicSymbolCode, IGeoStrokeStyle strokeStyle) {
114        int factor = 0;
115
116        switch (strokeStyle.getStrokePattern()) {
117            case solid:
118                factor = 0;
119                break;
120            case dashed:
121            case dotted: {
122                switch (basicSymbolCode) {
123                    case METOC_PRESSURE_INSTABILITY_LINE:
124                        factor = 2 * (int) strokeStyle.getStrokeWidth();
125                        break;
126                    case METOC_PRESSURE_SHEAR_LINE:
127                        factor = 2 * (int) strokeStyle.getStrokeWidth();
128                        break;
129                    case METOC_BOUNDED_AREAS_OF_WEATHER_LIQUID_PRECIPITATION_NON_CONVECTIVE_CONTINUOUS_OR_INTERMITTENT:
130                    case METOC_ATMOSPHERIC_BOUNDED_AREAS_OF_WEATHER_THUNDERSTORMS:
131                        factor = 3 * (int) strokeStyle.getStrokeWidth();
132                        break;
133                    default:
134                        // Normal dashes.
135                        factor = 3 * (int) strokeStyle.getStrokeWidth();
136                        break;
137                }
138                factor = (int) strokeStyle.getStrokeWidth();
139                break;
140            }
141        }
142
143        return factor;
144    }

146    public static short getTGStipplePattern(String basicSymbolCode, IGeoStrokeStyle strokeStyle) {
147        short pattern = 0;
148
149        switch (strokeStyle.getStrokePattern()) {
150            case solid:
151                pattern = 0;
152                break;
153            case dashed:
154            case dotted: {
155                switch (basicSymbolCode) {
156                    case METOC_PRESSURE_INSTABILITY_LINE:
157                        pattern = (short) 0xDFF6;
158                        break;
159                    case METOC_PRESSURE_SHEAR_LINE:
160                        pattern = (short) 0xFFF6;
161                        break;
162                    case METOC_BOUNDED_AREAS_OF_WEATHER_LIQUID_PRECIPITATION_NON_CONVECTIVE_CONTINUOUS_OR_INTERMITTENT:
163                    case METOC_ATMOSPHERIC_BOUNDED_AREAS_OF_WEATHER_THUNDERSTORMS:
164                        pattern = (short) 0xFFF6;
165                        break;
166                    default:
167                        // Normal dashes.
168                        pattern = (short) 0xEEEE;
169                        break;
170                }
171                break;
172            }
173        }
174
175        return pattern;
176    }
