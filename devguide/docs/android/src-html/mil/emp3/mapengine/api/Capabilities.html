001package mil.emp3.mapengine.api;
002
003import java.util.Arrays;
004import mil.emp3.api.enums.FeatureTypeEnum;
005import mil.emp3.api.enums.WMSVersionEnum;
006import mil.emp3.mapengine.interfaces.IMapEngineCapabilities;
007
008/**
009 * This class implements the IMapEngineCapabilities interface. It represent the 
010 * capabilities of a map engine which the core will constantly query. A map engine
011 * developer is free to implement its own class as long as the interface is implemented.
012 */
013public class Capabilities implements IMapEngineCapabilities {
014    private final java.util.Set<FeatureTypeEnum> canPlotSet;
015    private final java.util.Set<WMSVersionEnum> supportedWMSVersion;
016    
017    public Capabilities(FeatureTypeEnum[] aFeatureTypes, WMSVersionEnum[] aVersions) {
018        this.canPlotSet = new java.util.HashSet<>();
019        this.supportedWMSVersion = new java.util.HashSet<>();
020        
021        if (aFeatureTypes != null) {
022            this.canPlotSet.addAll(Arrays.asList(aFeatureTypes));
023        }
024        
025        if (aVersions != null) {
026            this.supportedWMSVersion.addAll(Arrays.asList(aVersions));
027        }
028    }
029
030    @Override
031    public boolean canPlot(FeatureTypeEnum eFeatureType) {
032        return this.canPlotSet.contains(eFeatureType);
033    }
034
035    @Override
036    public boolean wmsVersionSupported(WMSVersionEnum eVersion) {
037        return this.supportedWMSVersion.contains(eVersion);
038    }
039
040}
