038/**
039 * When a map engine is class loaded from another APK it is not able to access its resources and
040 * shared libraries. Class MapEngineContext was created to deal with the resource loading issue.
041 * It decorates the Context object and overrides getAssets and getResources methods to fetch resources
042 * from the map engine APK file.
043 *
044 * We have to resolve the issue of which API level this class consistent with.
045 */
046public class MapEngineContext extends Context {
047    private static final String TAG = MapEngineContext.class.getSimpleName();
048
049    private final Context context;
050    private Resources mResources;
051    private AssetManager mAssetManager;
052    private final String mDexPath;
053    private final String mPackageName;
054
055    public MapEngineContext(Context context, String applicationPackageName, String dexPath) {
056        this.context = context;
057        this.mDexPath = dexPath;
058        this.mPackageName = applicationPackageName;
059
060        Log.d(TAG, "applicationPackageName " + this.mPackageName + " dexPath " + this.mDexPath);
061        loadResources();
062/*
063        try {
064            mResources = context.getPackageManager().getResourcesForApplication(applicationPackageName);
065        } catch (PackageManager.NameNotFoundException e) {
066            e.printStackTrace();
067        }
068*/
069    }
070
071    protected void loadResources() {
072        try {
073            AssetManager assetManager = AssetManager.class.newInstance();
074            Method addAssetPath = assetManager.getClass().getMethod("addAssetPath", String.class);
075            addAssetPath.invoke(assetManager, mDexPath);
076            mAssetManager = assetManager;
077        } catch (Exception e) {
078            e.printStackTrace();
079        }
080        Resources superRes = context.getResources();
081        mResources = new Resources(mAssetManager, superRes.getDisplayMetrics(),
082                superRes.getConfiguration());
083        // mTheme = mResources.newTheme();
084        // mTheme.setTo(super.getTheme());
085    }
