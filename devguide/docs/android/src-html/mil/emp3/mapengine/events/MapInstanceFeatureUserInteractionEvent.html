001package mil.emp3.mapengine.events;
002
003import org.cmapi.primitives.IGeoPosition;
004
005import java.util.EnumSet;
006
007import mil.emp3.api.enums.UserInteractionKeyEnum;
008import mil.emp3.api.enums.UserInteractionMouseButtonEnum;
009import mil.emp3.api.interfaces.IFeature;
010import mil.emp3.api.enums.UserInteractionEventEnum;
011import mil.emp3.api.events.Event;
012import mil.emp3.mapengine.interfaces.IMapInstance;
013
014/**
015 * This class implements the Map Instance Feature User Interaction Event. It must be generated by all map engine
016 * instances when the user interacts with a feature. See {@link UserInteractionEventEnum}.
017 * All events of this type are handled by the EMP core code.
018 */
019public class MapInstanceFeatureUserInteractionEvent extends Event<UserInteractionEventEnum, IMapInstance> {
020    private final android.graphics.Point oPoint;
021    private final IGeoPosition oCoordinate;
022    private final IGeoPosition oStartCoordinate;
023    private final java.util.List<IFeature> oFeatureList;
024    private UserInteractionMouseButtonEnum oButton;
025    private final EnumSet<UserInteractionKeyEnum> oKeys;
026
027    // eventConsumed is used internally by MapInstanceEventHandler. It is looked at by the GestureHandler (WW PickNavigateController)
028    // for DRAG event only. Remember that DRAG and DRAG_COMPLETE events never propagate to applications. Applications should not
029    // use this member.
030    private boolean eventConsumed;
031    /**
032     * This constructor must be called by the map engines to creates a MapInstanceFeatureUserInteractionEvent event
033     * @param oMapInstance The actual map instance. The this property of the object that implements the IMapInstance interface.
034     * @param eEvent The enumerated value of the event that occurred. See {@link UserInteractionEventEnum}.
035     * @param featureList A list of references to the features on which the event occurred.
036     * @param oPointCoordinate The geographic coordinate of the location the event occurred.
037     * @param oPosition The cartesian coordinate relative to the origin of the view container.
038     * @param button Mouse button, left, right or middle
039     * @param keys Keyboard keys, shift, ctrl and/or alt
040     */
041    public MapInstanceFeatureUserInteractionEvent(IMapInstance oMapInstance,
042            UserInteractionEventEnum eEvent,
043            EnumSet<UserInteractionKeyEnum> keys,
044            UserInteractionMouseButtonEnum button,
045            java.util.List<IFeature> featureList,
046            android.graphics.Point oPointCoordinate,
047            IGeoPosition oPosition) {
048        super(eEvent, oMapInstance);
049        this.oPoint = oPointCoordinate;
050        this.oCoordinate = oPosition;
051        this.oStartCoordinate = null;
052        this.oFeatureList = featureList;
053        this.oButton = button;
054        this.oKeys = keys;
055    }
056
057    /**
058     * This constructor must be called by the map engines to creates a MapInstanceFeatureUserInteractionEvent event
059     * @param oMapInstance The actual map instance. The this property of the object that implements the IMapInstance interface.
060     * @param eEvent The enumerated value of the event that occurred. See {@link UserInteractionEventEnum}.
061     * @param featureList A list of references to the features on which the event occurred.
062     * @param oPointCoordinate The geographic coordinate of the location the event occurred.
063     * @param oPosition The cartesian coordinate relative to the origin of the view container.
064     * @param button Mouse button, left, right or middle
065     * @param keys Keyboard keys, shift, ctrl and/or alt
066     */
067    public MapInstanceFeatureUserInteractionEvent(IMapInstance oMapInstance,
068            UserInteractionEventEnum eEvent,
069            EnumSet<UserInteractionKeyEnum> keys,
070            UserInteractionMouseButtonEnum button,
071            java.util.List<IFeature> featureList,
072            android.graphics.Point oPointCoordinate,
073            IGeoPosition oPosition,
074            IGeoPosition oStartPosition) {
075        super(eEvent, oMapInstance);
076        this.oPoint = oPointCoordinate;
077        this.oCoordinate = oPosition;
078        this.oStartCoordinate = oStartPosition;
079        this.oFeatureList = featureList;
080        this.oButton = button;
081        this.oKeys = keys;
082    }
083
084    public android.graphics.Point getLocation() {
085        return this.oPoint;
086    }
087
088    public IGeoPosition getCoordinate() {
089        return this.oCoordinate;
090    }
091
092    public IGeoPosition getStartCoordinate() {
093        return this.oStartCoordinate;
094    }
095
096    public java.util.List<IFeature> getFeatures() {
097        return this.oFeatureList;
098    }
099
100    public UserInteractionMouseButtonEnum getButton() {
101        return this.oButton;
102    }
103
104    public EnumSet<UserInteractionKeyEnum> getKeys() {
105        return this.oKeys;
106    }
107
108    /**
109     * eventConsumed is used internally by MapInstanceEventHandler. It is looked at by the GestureHandler (WW PickNavigateController)
110     * for DRAG event only. Remember that DRAG and DRAG_COMPLETE events never propagate to applications. Applications should not
111     * use this method.
112     * @return true if event was consumed
113     */
114    public boolean isEventConsumed() {
115        return eventConsumed;
116    }
117
118    /**
119     * eventConsumed is used internally by MapInstanceEventHandler. It is looked at by the GestureHandler (WW PickNavigateController)
120     * for DRAG event only. Remember that DRAG and DRAG_COMPLETE events never propagate to applications. Applications should not
121     * use this method.
122     * @param eventConsumed
123     */
124    public void setEventConsumed(boolean eventConsumed) {
125        this.eventConsumed = eventConsumed;
126    }
127}
